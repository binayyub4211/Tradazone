
### 7. **Testing Script**

**scripts/test-linting.sh**:
```bash
#!/bin/bash
set -euo pipefail

echo "🧪 Testing CI Linting Configuration"
echo "================================"

# Colors for output
RED='\033[0;31m'
GREEN='\033[0;32m'
NC='\033[0m' # No Color

# Test 1: YAML validation
echo "📝 Test 1: Validating YAML files..."
if yamllint -c .yamllint.yml .github/workflows/ --strict; then
    echo -e "${GREEN}✅ YAML validation passed${NC}"
else
    echo -e "${RED}❌ YAML validation failed${NC}"
    exit 1
fi

# Test 2: Shell script linting
echo "📝 Test 2: Linting shell scripts..."
if find scripts/ -name "*.sh" -type f -exec shellcheck {} \; ; then
    echo -e "${GREEN}✅ ShellCheck passed${NC}"
else
    echo -e "${RED}❌ ShellCheck failed${NC}"
    exit 1
fi

# Test 3: GitHub Actions validation
echo "📝 Test 3: Validating GitHub Actions..."
if actionlint -color; then
    echo -e "${GREEN}✅ Actionlint passed${NC}"
else
    echo -e "${RED}❌ Actionlint failed${NC}"
    exit 1
fi

# Test 4: Verify no regressions in existing workflows
echo "📝 Test 4: Testing workflow execution simulation..."
python3 -c "
import yaml
import sys

workflows = ['.github/workflows/ci.yml', '.github/workflows/lint.yml']
for wf in workflows:
    try:
        with open(wf, 'r') as f:
            yaml.safe_load(f)
        print(f'✅ {wf} is valid')
    except Exception as e:
        print(f'❌ {wf} is invalid: {e}')
        sys.exit(1)
"

echo -e "${GREEN}✨ All tests passed!${NC}"
